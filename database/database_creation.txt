
// Database Creation
CREATE SCHEMA `stream_analysis`;

// user table
CREATE TABLE `stream_analysis`.`user_info` (
  `user_id` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(45) NOT NULL,
  `email` VARCHAR(60) NOT NULL,
  `password` VARCHAR(80) NOT NULL,
  PRIMARY KEY (`user_id`));
